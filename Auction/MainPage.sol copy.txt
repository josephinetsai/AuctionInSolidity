pragma solidity >=0.8.9;


contract MainPage{
    
    address[] public auctions;
    address[] public users;
        
    event ConstructAuction(address auctionContract, address owner, uint numAuctions, address[] allAuctions);
 
    function makeAuctionListing(uint bidIncrement, uint startBlock, uint endBlock) public{
        Auction newAuction = new Auction(msg.sender, bidIncrement, startBlock, endBlock);
        auctions.push(newAuction);
        users.push(msg.sender);

        emit ConstructAuction(newAuction, msg.sender, auctions.length, auctions);
    }
}
